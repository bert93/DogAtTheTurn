const postSchema = mongoose.Schema({
  title: String,
  content: String,
  date: Date,
  imageLocation: String,
  rating: {
      overall: Number,
      taste: Number,
      texture: Number,
      bun: Number,
      toppings: Number,
      service: Number
    },
  course: mongoose.Types.ObjectId
});

const Post = mongoose.model("post", postSchema);

const courseSchema = mongoose.Schema({
  _id: {
    type: mongoose.Types.ObjectId,
    ref: 'Post'
  },
  name: String,
  location: String,
  url: String
});
